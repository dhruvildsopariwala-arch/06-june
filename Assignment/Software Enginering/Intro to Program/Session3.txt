Question 1 : Declare variables for a Flipkart product: productName (as a string), price (float), and rating (double). Assign sample values and print each variable with its data type.
Answer: #include <stdio.h>

        int main() {
            char productName[] = "Iphone 17";
            int price = 80000;
            double rating = 4.7;

            printf("Product Name (string): %s\n", productName);
            printf("Price (int): %d\n", price);
            printf("Rating (double): %lf\n", rating);

            return 0;
        }
