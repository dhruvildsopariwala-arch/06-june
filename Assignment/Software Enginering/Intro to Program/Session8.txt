Question 1:Declare a function called getUserInitials that takes a user's full name (like 'Virat Kohli') and returns their initials in uppercase (e.g., 'VK'). Call this function with your favorite cricketer's name and print the result.
Answer:
#include <stdio.h>

void getUserInitials(char name[])
{
    printf("Initials: %c%c", name[0], name[7]);
}

int main()
{
    char name[] = "Virat Kohli";

    getUserInitials(name);

    return 0;
}

Question 2: Create a function addToCart that takes a shopping cart array and a product name, adds the product to the cart, and prints the updated cart. Demonstrate how passing the cart array by reference allows changes to persist outside the function.<br><br><em><strong>Hint:</strong> In languages like JavaScript, arrays are passed by reference. In C/C++, use pointers for reference behavior.</em>
Answer:
#include <stdio.h>
#include <string.h>

void addToCart(char cart[][20], int *count, char product[])
{
    strcpy(cart[*count], product);
    (*count)++;

    printf("\nCart Items:\n");
    for(int i = 0; i < *count; i++)
    {
        printf("%s\n", cart[i]);
    }
}

int main()
{
    char cart[5][20];
    int count = 0;

    addToCart(cart, &count, "Shoes");
    addToCart(cart, &count, "Watch");
    addToCart(cart, &count, "Bag");

    printf("\nCart in Main Function:\n");
    for(int i = 0; i < count; i++)
    {
        printf("%s\n", cart[i]);
    }

    return 0;
}

Question 3: Write two functions: increaseFollowersByValue and increaseFollowersByReference. Each should take a followers count and add 1000 to it, but one should use pass-by-value and the other pass-by-reference. Show how the original followers count changes (or doesn't) after calling each function.
Answer:
#include <stdio.h>

void value(int n)
{
    n = n + 1000;
    printf("Inside Value = %d\n", n);
}

void reference(int *n)
{
    *n = *n + 1000;
    printf("Inside Reference = %d\n", *n);
}

int main()
{
    int n = 5000;

    value(n);
    printf("Main = %d\n", n);

    reference(&n);
    printf("Main = %d\n", n);

    return 0;
}

Question 4: Build a function formatPrice that takes a price in rupees and returns a string formatted like Flipkart's price tag (e.g., '₹1,599'). Use this function to display prices for three different products.
Answer:
#include <stdio.h>

void formatPrice(int price)
{
    printf("₹1,%03d", price % 1000);
}

int main()
{
    printf("Mobile : ");
    formatPrice(1599);

    printf("\nHeadphones : ");
    formatPrice(2499);

    printf("\nKeyboard : ");
    formatPrice(3999);

    return 0;
}

Question 5: Refactor an existing function you wrote above to make it reusable for both product names and usernames (for example, a function that capitalizes the first letter of any string).<br><br><em><strong>Constraint:</strong> The refactored function should work for any string input, not just a specific use case.</em>
