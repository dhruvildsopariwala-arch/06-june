// Function to check if a number is even
function isEven(num)
{
    // Check if the number is divisible by 2
    if (num % 2 == 0)
    {
        // Number is even
        return true;
    }
    else
    {
        // Number is odd
        return false;
    }
}

Question 3:Write a function formatFollowersCount(count) that takes a number and returns a formatted string like Instagram: 1500 as '1.5K', 1200000 as '1.2M', and numbers below 1000 as-is. Add clear comments and use proper indentation.
Answer:
#include <stdio.h>

void formatFollowersCount(int count)
{
    // If followers are 10 lakh or more
    if(count >= 1000000)
    {
        printf("%.1fM", count / 1000000.0);
    }
    // If followers are 1000 or more
    else if(count >= 1000)
    {
        printf("%.1fK", count / 1000.0);
    }
    // If followers are less than 1000
    else
    {
        printf("%d", count);
    }
}

int main()
{
    formatFollowersCount(1500);
    printf("\n");

    formatFollowersCount(1200000);
    printf("\n");

    formatFollowersCount(850);

    return 0;
}
