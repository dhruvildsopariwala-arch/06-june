Question 1: Declare a string variable called songTitle and assign it the value 'Tum Hi Ho'. Print the length of the string using strlen().
Answer:
#include <stdio.h>
#include <string.h>

int main()
{
    char songTitle[] = "Tum Hi";

    printf("Song Title: %s\n", songTitle);
    printf("Length = %d", strlen(songTitle));

    return 0;
}
Question 2:Take input for two usernames (as strings) and compare them using strcmp(). Display whether they are the same or different.
Answer:
#include <stdio.h>
#include <string.h>

int main()
{
    char user1[20], user2[20];

    printf("Enter First Username: ");
    scanf("%s", user1);

    printf("Enter Second Username: ");
    scanf("%s", user2);

    if(strcmp(user1, user2) == 0)
    {
        printf("Both Usernames are Same");
    }
    else
    {
        printf("Both Usernames are Different");
    }

    return 0;
}

Question 4: Build a small script that takes the user's full name as input and creates a username by copying only the first 5 characters using strcpy(). Print the generated username.<br><br><em><strong>Constraint:</strong> If the name is shorter than 5 characters, use the full name as the username.</em>
Answer:
#include <stdio.h>
#include <string.h>

int main()
{
    char name[50];
    char username[20];

    printf("Enter Your Name: ");
    scanf("%s", name);

    if(strlen(name) < 5)
    {
        strcpy(username, name);
    }
    else
    {
        strncpy(username, name, 5);
        username[5] = '\0';
    }

    printf("Username: %s", username);

    return 0;
}
