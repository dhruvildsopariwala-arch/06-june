Question:
Use an AI tool of your choice (ChatGPT, Claude, GitHub Copilot, etc.) to help you write a C program
that:
Accepts exactly 10 integers from the user using a loop and stores them in an array.
Finds and displays the maximum value, minimum value, and arithmetic mean (displayed as a float with 2 decimal places).
Sorts the array in ascending order using any sorting method and displays the sorted list.
Prints whether the mean is closer to the minimum, closer to the maximum, or exactly midway
between them.
STEP 2 · TEST & DEBUG (WITHOUT AI)
Then, working without AI, test the code using boundary inputs: (a) all 10 values identical, (b) a
mix of positive and negative integers, and (c) a list where the mean falls exactly between min
and max. Find at least one bug, edge-case failure, or improvement in the AI's solution and fix it
yourself.
SUBMIT All 3 items required
1 The exact prompt(s) you gave the AI tool.
2 The AI's original code and your corrected version (submit both as separate .c files).
3 A 3–4 line note explaining what you changed and why the AI's original version needed
the correction.

Answer:
AI Code:
#include <stdio.h>

int main()
{
    int arr[10];
    int i, j;
    int max, min, temp;
    float mean, sum = 0;

    printf("Enter 10 integers:\n");

    for(i = 0; i < 10; i++)
    {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }

    max = min = arr[0];

    for(i = 1; i < 10; i++)
    {
        if(arr[i] > max)
            max = arr[i];

        if(arr[i] < min)
            min = arr[i];
    }

    mean = sum / 10;

    for(i = 0; i < 9; i++)
    {
        for(j = 0; j < 9 - i; j++)
        {
            if(arr[j] > arr[j + 1])
            {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    printf("\nMinimum = %d", min);
    printf("\nMaximum = %d", max);
    printf("\nMean = %.2f\n", mean);

    printf("\nSorted Array: ");

    for(i = 0; i < 10; i++)
        printf("%d ", arr[i]);

    if((mean - min) < (max - mean))
        printf("\nMean is closer to Minimum");
    else if((mean - min) > (max - mean))
        printf("\nMean is closer to Maximum");
    else
        printf("\nMean is exactly midway");

    return 0;
}

#include <stdio.h>

int main()
{
    int a[10];
    int i, j, temp;
    int max, min;
    float sum = 0, avg;

    printf("Enter 10 numbers:\n");

    for(i = 0; i < 10; i++)
    {
        scanf("%d",&a[i]);
        sum = sum + a[i];
    }

    max = a[0];
    min = a[0];

    for(i = 1; i < 10; i++)
    {
        if(a[i] > max)
            max = a[i];

        if(a[i] < min)
            min = a[i];
    }

    avg = sum / 10;

    for(i = 0; i < 9; i++)
    {
        for(j = i + 1; j < 10; j++)
        {
            if(a[i] > a[j])
            {
                temp = a[i];
                a[i] = a[j];
                a[j] = temp;
            }
        }
    }

    printf("\nMin = %d", min);
    printf("\nMax = %d", max);
    printf("\nAverage = %.2f\n", avg);

    printf("\nSorted Numbers:\n");

    for(i = 0; i < 10; i++)
    {
        printf("%d ", a[i]);
    }

    float d1 = avg - min;
    float d2 = max - avg;

    if(d1 < 0) d1 = -d1;
    if(d2 < 0) d2 = -d2;

    if(d1 < d2)
        printf("\nAverage is closer to Minimum");
    else if(d2 < d1)
        printf("\nAverage is closer to Maximum");
    else
        printf("\nAverage is exactly midway");

    return 0;
}

Explaining Difference:
I changed the comparison used to find whether the average is closer to the minimum or maximum. The original AI code compared values directly, which could give incorrect results in some cases. I used absolute differences before comparing the distances. I also changed the sorting loop to a simpler method that I understood better.
